void InitialisationM();


Mix_Music* loadM(char* p);
	Permet de charger une musique avec une adresse d'un élément d'extention .mp3

void jouerM(Mix_Music* musique,int tour);
	Permet de jouer une musique pendant un nombre de tour
	(-1 tour => répétition infini)

void erreurmusique(const char * message,Mix_Music* musique);
	Permet de gérer les erreur liées à la musique

void erreurchunk(const char * message,Mix_Chunk* chunk);
	Permet de gérer les erreur liées aux chunk

void initialisationCanal(int canal);
	Permet d'initialiser des canaux de sons

Mix_Chunk * loadChunk(char * p);
	Permet de charger un chunk avec une adresse d'un élément d'extention .wav


void operationMusique(int op,int volume);
	Permet selon la valeur de op de réaliser une opération sur la musique:
	-0 met en pause ou enlève la pause
	-1 rest la musique
	-2 arrête la musique 
	La variable volume est le volume associé à la musique
	-3 diminue le volume
	-4 augmente le volume
